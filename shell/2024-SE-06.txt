#!/bin/bash

if [[ "${#}" -ne 1 ]]; then
  echo "One needed arg"
  exit 1
fi

if [[ ! -f "${1}" ]]; then
  echo "Arg should ne a file"
  exit 2
fi

paths=$(cat "${1}" | awk '{print $1}' | sort | uniq)
rows=$(mktemp)
while read line; do
  last_line=$(grep -E "^${line}" "${1}" | tail -n 1)
  echo "${last_line}" >> "${rows}"
done< <(echo "${paths}")

while read line; do
  path=$(echo "${line}" | awk '{print $1}')
  file_type=$(echo "${line}" | awk '{print $2}')
  mode=$(echo "${line}" | awk '{print $3}')
  perm=$(echo "${line}" | awk '{print $4}')

  if [[ "${file_type}" == "dir" ]]; then
    command="-d"
  elif [[ "${file_type}" == "file" ]]; then
    command="-f"
  elif [[ "${file_type}" == "symlink" ]]; then
    command="-h"
  elif [[ "${file_type}" == "nonexistant" ]]; then
    command="! -e"
  else
    continue
  fi

  if [[ -e "${path}" && ! ${command} "${path}" ]]; then
    rm "${path}" 2>/dev/null
    if [[ "${file_type}" == "dir" ]]; then
      mkdir -p "${path}"
    elif [[ "${file_type}" == "file" ]]; then
      mkdir -p "$(dirname "${path}")"
      touch "${path}"
    elif [[ "${file_type}" == "symlink" ]]; then
      ln -s "${perm}" "${path}"
    fi
    chmod "${perm}" "${path}"
    [[ !-z "${mode}" ]] && chown "${mode}" "${path}"
  elif [[ -e "${path}" && ${command} "${path}" ]]; then
    chmod "${perm}" "${path}"
    [[ !-z "${mode}" ]] && chown "${mode}" "${path}"
  elif [[ ! -e "${path}" ]]; then
    dir=$(dirname "${path}")
    [[ ! -d "${dir}" ]] && mkdir -p "${dir}" -m $(umask)
    touch "${path}"
  fi
done< <(cat "${rows}")
